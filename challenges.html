//This is the beginning of the Math functions
function SolveMath() {
 
    //This assigns the input numbers to variables
    let num1 = parseInt(document.getElementById("num1").value);
    let num2 = parseInt(document.getElementById("num2").value);
    let num3 = parseInt(document.getElementById("num3").value);
    let num4 = parseInt(document.getElementById("num4").value);
    let num5 = parseInt(document.getElementById("num5").value);
 
    //Check to ensure all numbers entered are between 1 and 100
    if (num1 > 100 || num2 > 100 || num3 > 100 || num4 > 100 || num5 > 100) {
        alert("Please enter a number between 1 and 100." );
        ClearMathData();
        return;
    }
    if (num1 < 1 || num2 < 1 || num3 < 1 || num4 < 1 || num5 < 1) {
        alert("Please enter a number between 1 and 100." );
        ClearMathData();
        return;
    }
    //Check to ensure user enter numbers
    if (isNaN(num1) || isNaN(num2) || isNaN(num3) || isNaN(num4) || isNaN(num5)) {
        alert("Please enter a number between 1 and 100." );
        ClearMathData();
        return;
    }
 
    //Perform math operations
    var min = Math.min(num1, num2, num3, num4, num5);
    var max = Math.max(num1, num2, num3, num4, num5);
    var sum = num1 + num2 + num3 + num4 + num5;
    var mean = sum / 5;
    var product = num1 * num2 * num3 * num4 * num5;
 
    //Display math results
    document.getElementById('min').innerHTML = "The smallest number is " + min;
    document.getElementById('max').innerHTML = "The largest number is " + max;
    document.getElementById('sum').innerHTML = "The sum of the numbers is " + sum;
    document.getElementById('mean').innerHTML = "The mean is " + mean;
    document.getElementById('product').innerHTML = "The product is " + product;
 
}
